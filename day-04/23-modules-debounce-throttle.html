JS Modules (import/export) & Performance Techniques

1. ES6 Modules - export

// math.js - named exports
export const add = (a, b) => a + b
export const subtract = (a, b) => a - b
export const PI = 3.14159

// utils.js - default export
export default function greet(name) {
  return `Hello, ${name}!`
}

2. ES6 Modules - import

// Import named exports
import { add, subtract, PI } from './math.js'
console.log(add(5, 3))     // 8
console.log(PI)            // 3.14159

// Import default export
import greet from './utils.js'
console.log(greet('John')) // Hello, John!

// Import everything
import * as math from './math.js'
console.log(math.add(2, 3))

3. Using Modules in HTML

<!-- Need type="module" in script tag -->
<script type="module">
  import { add } from './math.js'
  console.log(add(10, 20))
</script>

<!-- Or link to external module file -->
<script type="module" src="app.js"></script>

4. Debounce - Wait until user stops typing

function debounce(func, delay = 300) {
  let timeoutId
  return function (...args) {
    clearTimeout(timeoutId)
    timeoutId = setTimeout(() => func.apply(this, args), delay)
  }
}

// Usage: search input that waits for user to stop typing
const searchInput = document.getElementById('search')
const handleSearch = debounce((event) => {
  console.log('Searching:', event.target.value)
  // Make API call here
}, 500)

searchInput.addEventListener('input', handleSearch)

When to use debounce: Search inputs, form auto-save, window resize

5. Throttle - Limit how often a function runs

function throttle(func, limit = 200) {
  let inThrottle
  return function (...args) {
    if (!inThrottle) {
      func.apply(this, args)
      inThrottle = setTimeout(() => (inThrottle = false), limit)
    }
  }
}

// Usage: scroll handler that runs max once every 200ms
const handleScroll = throttle(() => {
  console.log('Scrolled at:', new Date().toISOString())
}, 200)

window.addEventListener('scroll', handleScroll)

When to use throttle: Scroll events, resize events, button click spam prevention
