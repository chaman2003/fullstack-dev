Day 11: Component Architecture & Design Patterns

1. Smart vs Dumb Components

Smart (Container): Has state, handles logic, fetches data

Dumb (Presentational): Just displays props, no logic, reusable

// Smart Component
function UsersPage() {
  const [users, setUsers] = useState([]);

  useEffect(() => {
    fetch('/api/users')
      .then(r => r.json())
      .then(setUsers);
  }, []);

  return <UserList users={users} />;
}

// Dumb Component
function UserList({ users }) {
  return (
    <ul>
      {users.map(user => (
        <li key={user.id}>{user.name}</li>
      ))}
    </ul>
  );
}

2. Higher-Order Component (HOC)

A function that takes a component and returns a new component with extra features.

// HOC - adds loading state to any component
function withLoading(Component) {
  return function WithLoading({ isLoading, ...props }) {
    if (isLoading) return <div>Loading...</div>;
    return <Component {...props} />;
  };
}

// Usage
const UserListWithLoading = withLoading(UserList);

function App() {
  const [loading, setLoading] = useState(true);
  const [users, setUsers] = useState([]);

  useEffect(() => {
    fetch('/api/users')
      .then(r => r.json())
      .then(data => { setUsers(data); setLoading(false); });
  }, []);

  return <UserListWithLoading isLoading={loading} users={users} />;
}

3. Reusable Components

// Reusable Button
function Button({ children, variant = 'primary', onClick }) {
  const styles = {
    primary: { background: 'blue', color: 'white' },
    secondary: { background: 'gray', color: 'white' },
    danger: { background: 'red', color: 'white' },
  };

  return (
    <button
      onClick={onClick}
      style={{
        ...styles[variant],
        padding: '10px 20px',
        border: 'none',
        borderRadius: '5px',
        cursor: 'pointer',
      }}
    >
      {children}
    </button>
  );
}

// Usage
<Button variant="primary" onClick={() => alert("Clicked")}>Save</Button>
<Button variant="danger" onClick={() => alert("Deleted")}>Delete</Button>
